179// imm p2p sdk 依赖几个外部服务，通过回调或接口实现：
180// 1. 信令发送
181// 当 imm p2p sdk 需要发送信令时，会调用上层实现的 imm_p2p_rtc_signaling_cb_t 进行发送
182// 2. 信令接收
183// 当上层收到信令时，通过 imm_p2p_rtc_set_signaling 接口设置到 imm p2p sdk
184// 3. HTTP 服务
185// 当 imm p2p sdk 需要发送日志时，会调用上层实现的 imm_p2p_rtc_log_cb_t 进行发送
186
187// 信令回调，用于发送信令
188// remote_id: 表示向 remote_id 发送信令
189// signaling: 信令内容的起始地址
190// len: 信令内容的长度
191typedef void (*imm_p2p_rtc_signaling_cb_t)(char *remote_id, char *signaling, uint32_t len);
192
193// 日志回调，用于发送日志
194// log: 日志内容的起始地址
195// len: 日志内容的长度
196// flags: 0-不上传云端，1-上传云端
197typedef void (*imm_p2p_rtc_log_cb_t)(char *log, uint32_t len, uint32_t flags);
198
199// 鉴权回调，用于验证offer是否合法
200// 用 hmac-sha256 对 buf 内容计算 hash，再对 hash 结果计算 base64 编码，
201// 计算方法：base64(hmac-sha256(key=password, content=buf, length=len))
202// 将结果跟 md 比较, 相同返回 0， 不同返回 -1
203typedef int32_t (*imm_p2p_rtc_auth_cb_t)(char *buf, uint32_t len, char *md, uint32_t md_len);
204
205// http回调，用于发送http请求

207// devId: 设备编号
208// content: 请求内容
209// content_len: content长度
210typedef int32_t (*imm_p2p_rtc_http_cb_t)(char *api, char *devId, char *content, uint32_t content_len);
