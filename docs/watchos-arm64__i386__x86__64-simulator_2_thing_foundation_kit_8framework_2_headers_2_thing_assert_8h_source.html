29// 检查是否处于被 Xcode 调试中
30// 参考 https://developer.apple.com/library/archive/qa/qa1361/_index.html
31static inline bool AmIBeingDebugged(void)
32 // Returns true if the current process is being debugged (either
33 // running under the debugger or has a debugger attached post facto).
34{
35 int junk;
36 int mib[4];
37 struct kinfo_proc info;
38 size_t size;
39
40 // Initialize the flags so that, if sysctl fails for some bizarre
41 // reason, we get a predictable result.
42
43 info.kp_proc.p_flag = 0;
44
45 // Initialize mib, which tells sysctl the info we want, in this case
46 // we're looking for information about a specific process ID.
47
48 mib[0] = CTL_KERN;
49 mib[1] = KERN_PROC;
50 mib[2] = KERN_PROC_PID;
51 mib[3] = getpid();
52
53 // Call sysctl.
54
55 size = sizeof(info);
56 junk = sysctl(mib, sizeof(mib) / sizeof(*mib), &info, &size, NULL, 0);
57 assert(junk == 0);
58
59 // We're being debugged if the P_TRACED flag is set.
60
61 return ( (info.kp_proc.p_flag & P_TRACED) != 0 );
62}
